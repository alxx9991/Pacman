    public static ArrayList<char[]> readMapFile(String filename) {
        try {
            ArrayList<char[]> grid = new ArrayList<char[]>();
            File file = new File(filename);
            Scanner scanner = new Scanner(file);
        
            while (scanner.hasNextLine()) {
                String line = scanner.next();
                char[] lineList = new char[line.length()];
                for (int i = 0; i < line.length(); i++) {
                    lineList[i] = line.charAt(i);
                }
                grid.add(lineList);
            }
            scanner.close();
            return grid;
        } catch (FileNotFoundException e) {
            return null;
        }
    }

    public void generateObjects() {
        int x = 0;
        int y = 0;

        for (char[] i: this.grid) {
            for (char j: i) {
                if (Integer.valueOf(j) >= 1 && Integer.valueOf(j) <= 6) {
                    this.app.objects.add(new Wall(x, y, j, this.app));
                } else if (j == '7') {
                    this.app.objects.add(new Fruit(x, y, this.app.loadImage("fruit.png"), this.app));
                } else {
                    ;
                }
                x += 16;
            }
            y += 16;
        }
    }
}
