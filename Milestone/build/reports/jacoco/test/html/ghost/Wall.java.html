package ghost;

import processing.core.PImage;
import processing.core.PApplet;

import java.util.HashMap;;

public class Wall extends GameObject {
    char type;
    App app;
    

    public Wall(int x, int y, char type, App app) {
        this.x = x;
        this.y = y;
        this.type = type;
        this.app = app;
        
        switch (type) {
            case 1: 
                this.sprite = app.loadImage("horizontal.png");
                break;
            case 2: 
                this.sprite = app.loadImage("vertical.png");
                break;
            case 3: 
                this.sprite = app.loadImage("upLeft.png");
                break;
            case 4: 
                this.sprite = app.loadImage("upRight.png");
                break;
            case 5: 
                this.sprite = app.loadImage("downLeft.png");
                break;
            case 6: 
                this.sprite = app.loadImage("downRight.png");
                break;
        }
    }
}
